// Generate sections dynamically
function generateSections() {
    const contentArea = document.getElementById('content-area');
    
    if (!contentArea) {
        console.error('Content area not found!');
        return;
    }
    
    console.log('Found content area, generating', sectionData.length, 'sections');
    
    // Generate content sections
    sectionData.forEach((section, index) => {
        try {
            const sectionDiv = document.createElement('div');
            sectionDiv.className = 'collapsible-section';
            
            const listItemsHtml = section.listItems.map(item => `<li>${item}</li>`).join('');
            
            // Determine if we should use video or image
            const isVideo = section.imageUrl.toLowerCase().endsWith('.mov') || 
                          section.imageUrl.toLowerCase().endsWith('.mp4') || 
                          section.imageUrl.toLowerCase().endsWith('.webm');
            
            const mediaHtml = isVideo ? 
                `<video src="${section.imageUrl}" alt="${section.imageAlt}" autoplay loop muted playsinline></video>` :
                `<img src="${section.imageUrl}" alt="${section.imageAlt}">`;
            
            sectionDiv.innerHTML = `
                <button class="section-header" id="${section.id}" onclick="toggleSection('${section.id}')">
                    <span>${section.title}</span>
                    <span class="section-icon">▼</span>
                </button>
                <div class="section-content" id="${section.id}-content">
                    <div class="content-inner">
                        <div class="gif-container">
                            ${mediaHtml}
                        </div>
                        <div class="text-container">
                            <h4>${section.contentTitle}</h4>
                            <p>${section.description}</p>
                            <ul>
                                ${listItemsHtml}
                            </ul>
                            <p>${section.conclusion}</p>
                        </div>
                    </div>
                </div>
            `;
            
            contentArea.appendChild(sectionDiv);
            console.log('Generated section:', section.id);
        } catch (error) {
            console.error('Error generating section', index, ':', error);
        }
    });
}

let currentActiveSection = null;

// Function called by section headers - toggles sections
function toggleSection(sectionId) {
    const header = document.getElementById(sectionId);
    const content = document.getElementById(sectionId + '-content');
    const icon = header.querySelector('.section-icon');
    const isActive = content.classList.contains('active');
    
    if (isActive) {
        // Close the section
        closeSection(sectionId);
    } else {
        // Close any other section first
        if (currentActiveSection && currentActiveSection !== sectionId) {
            closeSection(currentActiveSection);
        }
        
        // Open this section
        header.classList.add('active');
        content.classList.add('active');
        icon.style.transform = 'rotate(180deg)';
        currentActiveSection = sectionId;
    }
}

// Helper function to close a section
function closeSection(sectionId) {
    const header = document.getElementById(sectionId);
    const content = document.getElementById(sectionId + '-content');
    const icon = header.querySelector('.section-icon');
    
    header.classList.remove('active');
    content.classList.remove('active');
    icon.style.transform = 'rotate(0deg)';
    
    if (currentActiveSection === sectionId) {
        currentActiveSection = null;
    }
}
